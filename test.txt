        async function initializePage() {
            const date = getDateFromFilename();
            
            if (!date) {
                showError('No date specified in URL');
                return;
            }

            const formattedDate = formatDateForDisplay(date);
            updatePageContent(date, formattedDate);

                // Wait for database to be ready
                if (window.waitForSupabase) {
                    await window.waitForSupabase();
                }

                // Load puzzle data from database
                if (window.PuzzleDatabase) {
                    const puzzleData = await window.PuzzleDatabase.getPuzzleByDate(date);
                    if (puzzleData) {
                        // Set up puzzle data for the game engine
                        window.PUZZLE_DATA = puzzleData;
                        window.PUZZLE_DATE = date;
                        
                        
                        // Initialize the game and remove loading state
                        if (typeof NYTArchiveGame === 'undefined') {
                            throw new Error('NYTArchiveGame class not loaded');
                        }
                        window.nytGame = new NYTArchiveGame(window.PUZZLE_DATA, window.PUZZLE_DATE);
                        
                        // Add sound functionality to the game
                        window.nytGame.soundEnabled = soundEnabled;
                        window.nytGame.clickSound = clickSound;
                        window.nytGame.playSound = playSound;
                        
                        // Override the rotateDomino method to add sound

                        window.nytGame.rotateDomino = function(dominoIndex) {
                            originalRotateDomino(dominoIndex);
                            playSound('click');
                        };
                        
                        // Remove loading spinner - just remove the class, let the game handle the content
                        const gameContainer = document.getElementById('game-board-container');
                        if (gameContainer) {
                            gameContainer.classList.remove('loading');
                        }
                    } else {
                        throw new Error(`No puzzle data found for ${date}`);
                    }
                } else {
                    throw new Error('Database connection not available');
                }

                showError(`Unable to load puzzle data for ${formattedDate}. ${error.message}`);
            }

                border-radius: 6px; font-size: 0.9em; opacity: 0;
                transition: opacity 0.3s ease;
            `;
            notification.textContent = `Switching to ${theme} theme...`;
            document.body.appendChild(notification);
            
            setTimeout(() => notification.style.opacity = '1', 10);
            
            // Hard refresh the page after a brief delay to show notification
            setTimeout(() => {
                window.location.reload();
            }, 800);
        }

        function loadSavedTheme() {
            const savedTheme = localStorage.getItem('colorTheme') || 'gelato';
            const themeSelector = document.getElementById('themeSelector');
            if (themeSelector) {
                themeSelector.value = savedTheme;
            }
        }

        function updateSoundToggle() {
            const toggle = document.getElementById('soundToggle');
            if (toggle) {
                toggle.checked = soundEnabled;
            }
        }

        // Global function for timer toggle button
        function toggleTimer() {
            if (window.nytGame && typeof window.nytGame.toggleTimer === 'function') {
                window.nytGame.toggleTimer();
            }
        }

        // Initialize page when DOM is loaded
        document.addEventListener('DOMContentLoaded', function() {
            initAudio();
            updateSoundToggle();
            loadSavedTheme();
            initializePage();
        });
    </script>
